#!/usr/bin/python3
matrix_divided = __import__('2-matrix_divided').matrix_divided

# Test 1: Normal case
matrix = [
    [1, 2, 3],
    [4, 5, 6]
]
print(matrix_divided(matrix, 3))

# Test 2: Division by a float
matrix2 = [
    [9, 12, 15],
    [18, 21, 24]
]
print(matrix_divided(matrix2, 4.5))

# Test 3: Error if div = 0
try:
    print(matrix_divided(matrix, 0))
except Exception as e:
    print(e)

# Test 4: Error if div is not a number
try:
    print(matrix_divided(matrix, "a"))
except Exception as e:
    print(e)

# Test 5: Error if the matrix contains something other than int/float
bad_matrix = [
    [1, 2, "3"],
    [4, 5, 6]
]
try:
    print(matrix_divided(bad_matrix, 2))
except Exception as e:
    print(e)

# Test 6: Error if rows have different sizes
bad_matrix2 = [
    [1, 2],
    [3, 4, 5]
]
try:
    print(matrix_divided(bad_matrix2, 2))
except Exception as e:
    print(e)

# Test 7: Empty matrix
print(matrix_divided([], 2))

# Test 8: Matrix with an empty row
print(matrix_divided([[]], 2))
